public class tablero{
    private int casillas;
    private String[] fichas;

    Ficha posicion= new Ficha(null);


    public tablero(){
        initfichas();
        casillas=10;
    }

    private void initfichas(){
        fichas= new String[2];
        fichas[0]="Verde";
        fichas[1]="Azul";
    }
    public void juego(){
        int i=0;
        while(i<fichas.length){
            int avanza=posicion.avanzar();
            if (avanza>=casillas){
                i+=1;
                System.out.println("Ha ganado la ficha: "+fichas[i]);
            }
            else{
                i+=1;
                System.out.println("Ha ganado la ficha: "+fichas[i]);
            }
            
        }

    }
}
