public class Ficha {
    public String color;
    public int posicion;
    
    Dado dado = new Dado();

    public Ficha(String colorFicha){
        color=colorFicha;
        posicion=0;
        
    }
    public int avanzar(){
        int pasos=dado.lanzar();
        posicion+=pasos;
         
        return posicion;
    }
    }
