public class Ficha {
    private String color;
    private int posicion;
    private int dado;

    public Ficha(){
        color=" ";
        posicion=0;
        dado=(int)(Math.random()*6+1);
    }
    public int avanzar(){
        int pasos=dado;
        posicion+=pasos;
        return posicion;      
    }
    }
